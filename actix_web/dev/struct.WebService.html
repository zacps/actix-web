fn index(req: dev::ServiceRequest) -> dev::ServiceResponse {
    req.into_response(HttpResponse::Ok().finish())
}

fn main() {
    let app = App::new()
        .service(
            web::service("/app")
                .guard(guard::Header("content-type", "text/plain"))
                .finish(index)
        );
}
